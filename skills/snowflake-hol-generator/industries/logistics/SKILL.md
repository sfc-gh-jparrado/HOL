CREATE OR REPLACE TABLE RAW.VEHICULOS (
    ID_VEHICULO VARCHAR(10) PRIMARY KEY,
    PLACA VARCHAR(10),
    TIPO VARCHAR(30), -- TRAILER, TORTON, CAMIONETA, MOTO, BICICLETA
    MARCA VARCHAR(30),
    MODELO VARCHAR(30),
    ANIO NUMBER(4),
    CAPACIDAD_KG NUMBER,
    CAPACIDAD_PAQUETES NUMBER,
    ID_CEDIS_BASE VARCHAR(10),
    ESTATUS VARCHAR(20), -- DISPONIBLE, EN_RUTA, MANTENIMIENTO, BAJA

    FOREIGN KEY (ID_CEDIS_BASE) REFERENCES RAW.CENTROS_DISTRIBUCION(ID_CEDIS)

INSERT INTO RAW.VEHICULOS
SELECT 
    'VEH' || LPAD(SEQ4()::VARCHAR, 4, '0') AS ID_VEHICULO,
    UPPER(SUBSTRING(MD5(RANDOM()::VARCHAR), 1, 3)) || '-' || UNIFORM(100, 999, RANDOM()) || '-' || 
    UPPER(SUBSTRING(MD5(RANDOM()::VARCHAR), 1, 1)) AS PLACA,
    ARRAY_CONSTRUCT('TRAILER', 'TORTON', 'TORTON', 'CAMIONETA', 'CAMIONETA', 'CAMIONETA', 'MOTO', 'MOTO', 'BICICLETA')[UNIFORM(0, 8, RANDOM())]::VARCHAR AS TIPO,

    '202' || UNIFORM(0, 4, RANDOM()) AS MODELO,
    UNIFORM(2019, 2024, RANDOM()) AS ANIO,
    CASE 
        WHEN UNIFORM(0, 8, RANDOM()) < 2 THEN 30000
        WHEN UNIFORM(0, 8, RANDOM()) < 4 THEN 10000
        WHEN UNIFORM(0, 8, RANDOM()) < 7 THEN 1500

    ARRAY_CONSTRUCT('DISPONIBLE', 'DISPONIBLE', 'DISPONIBLE', 'EN_RUTA', 'EN_RUTA', 'MANTENIMIENTO')[UNIFORM(0, 5, RANDOM())]::VARCHAR AS ESTATUS,

## Vistas Analíticas

```sql
-- Vista de KPIs operativos
CREATE OR REPLACE VIEW ANALYTICS.V_KPIS_OPERATIVOS AS
SELECT 
    DATE_TRUNC('DAY', e.FECHA_CREACION) AS DIA,
    e.TIPO_SERVICIO,
    c.NOMBRE AS CEDIS_ORIGEN,
    
    -- Volumen
    COUNT(*) AS TOTAL_ENVIOS,
    SUM(e.PESO_KG) AS PESO_TOTAL_KG,
    
    -- Entregas
    COUNT(CASE WHEN e.ESTATUS = 'ENTREGADO' THEN 1 END) AS ENTREGADOS,
    COUNT(CASE WHEN e.ESTATUS = 'INTENTO_FALLIDO' THEN 1 END) AS INTENTOS_FALLIDOS,
    COUNT(CASE WHEN e.ESTATUS = 'DEVOLUCION' THEN 1 END) AS DEVOLUCIONES,
    
    -- KPIs
    ROUND(COUNT(CASE WHEN e.ESTATUS = 'ENTREGADO' THEN 1 END)::FLOAT / NULLIF(COUNT(*), 0) * 100, 2) AS TASA_ENTREGA_PCT,
    ROUND(COUNT(CASE WHEN e.FECHA_ENTREGA_REAL <= e.FECHA_ESTIMADA_ENTREGA THEN 1 END)::FLOAT / 
          NULLIF(COUNT(CASE WHEN e.ESTATUS = 'ENTREGADO' THEN 1 END), 0) * 100, 2) AS ON_TIME_DELIVERY_PCT,
    ROUND(AVG(e.INTENTOS_ENTREGA), 2) AS PROMEDIO_INTENTOS

FROM RAW.ENVIOS e
JOIN RAW.CENTROS_DISTRIBUCION c ON e.ID_CEDIS_ORIGEN = c.ID_CEDIS
GROUP BY DATE_TRUNC('DAY', e.FECHA_CREACION), e.TIPO_SERVICIO, c.NOMBRE;

-- Vista de performance de conductores
CREATE OR REPLACE VIEW ANALYTICS.V_PERFORMANCE_CONDUCTORES AS
SELECT 
    c.ID_CONDUCTOR,
    c.NOMBRE,
    cd.NOMBRE AS CEDIS_BASE,
    c.CALIFICACION,
    c.ENTREGAS_A_TIEMPO_PCT,
    
    -- Métricas de período
    COUNT(e.ID_ENVIO) AS ENVIOS_ASIGNADOS,
    COUNT(CASE WHEN e.ESTATUS = 'ENTREGADO' THEN 1 END) AS ENVIOS_ENTREGADOS,
    ROUND(AVG(e.PESO_KG), 2) AS PESO_PROMEDIO,
    
    -- Clasificación
    CASE 
        WHEN c.CALIFICACION >= 4.5 AND c.ENTREGAS_A_TIEMPO_PCT >= 95 THEN '⭐ Excelente'
        WHEN c.CALIFICACION >= 4.0 AND c.ENTREGAS_A_TIEMPO_PCT >= 90 THEN '✅ Bueno'
        WHEN c.CALIFICACION >= 3.5 THEN '⚠️ Regular'
        ELSE '🔴 Requiere atención'
    END AS CLASIFICACION_PERFORMANCE

FROM RAW.CONDUCTORES c
JOIN RAW.CENTROS_DISTRIBUCION cd ON c.ID_CEDIS_BASE = cd.ID_CEDIS
LEFT JOIN RAW.ENVIOS e ON c.ID_CONDUCTOR = e.ID_CONDUCTOR
WHERE c.ESTATUS = 'ACTIVO'
GROUP BY c.ID_CONDUCTOR, c.NOMBRE, cd.NOMBRE, c.CALIFICACION, c.ENTREGAS_A_TIEMPO_PCT;

-- Vista de utilización de flota
CREATE OR REPLACE VIEW ANALYTICS.V_UTILIZACION_FLOTA AS
SELECT 
    v.TIPO,
    cd.NOMBRE AS CEDIS_BASE,
    COUNT(*) AS TOTAL_VEHICULOS,
    COUNT(CASE WHEN v.ESTATUS = 'DISPONIBLE' THEN 1 END) AS DISPONIBLES,
    COUNT(CASE WHEN v.ESTATUS = 'EN_RUTA' THEN 1 END) AS EN_RUTA,
    COUNT(CASE WHEN v.ESTATUS = 'MANTENIMIENTO' THEN 1 END) AS EN_MANTENIMIENTO,
    ROUND(COUNT(CASE WHEN v.ESTATUS = 'EN_RUTA' THEN 1 END)::FLOAT / NULLIF(COUNT(*), 0) * 100, 2) AS UTILIZACION_PCT,
    ROUND(AVG(v.KM_ACUMULADOS), 0) AS KM_PROMEDIO,
    AVG(DATEDIFF('day', v.ULTIMO_SERVICIO, CURRENT_DATE())) AS DIAS_DESDE_SERVICIO
FROM RAW.VEHICULOS v
JOIN RAW.CENTROS_DISTRIBUCION cd ON v.ID_CEDIS_BASE = cd.ID_CEDIS
WHERE v.ESTATUS != 'BAJA'
GROUP BY v.TIPO, cd.NOMBRE;
```
