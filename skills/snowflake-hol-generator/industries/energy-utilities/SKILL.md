## Datos Sintéticos

### Tabla: CLIENTES_UTILITY
```sql
CREATE OR REPLACE TABLE RAW.CLIENTES (
    ID_CLIENTE VARCHAR(15) PRIMARY KEY,
    NUMERO_CUENTA VARCHAR(20),
    TIPO_CLIENTE VARCHAR(20), -- RESIDENCIAL, COMERCIAL, INDUSTRIAL, GOBIERNO
    NOMBRE VARCHAR(200),
    DIRECCION VARCHAR(300),
    CIUDAD VARCHAR(50),
    CODIGO_POSTAL VARCHAR(10),
    TARIFA VARCHAR(20),
    FECHA_ALTA DATE,
    MEDIDOR_ID VARCHAR(20),
    TIPO_MEDIDOR VARCHAR(20), -- TRADICIONAL, SMART
    ESTATUS VARCHAR(20)
);

-- Generar clientes
INSERT INTO RAW.CLIENTES
SELECT 
    'CLI' || LPAD(SEQ4()::VARCHAR, 8, '0') AS ID_CLIENTE,
    LPAD(UNIFORM(100000000000, 999999999999, RANDOM())::VARCHAR, 12, '0') AS NUMERO_CUENTA,
    ARRAY_CONSTRUCT('RESIDENCIAL', 'RESIDENCIAL', 'RESIDENCIAL', 'RESIDENCIAL', 'COMERCIAL', 'COMERCIAL', 'INDUSTRIAL', 'GOBIERNO')[UNIFORM(0, 7, RANDOM())]::VARCHAR AS TIPO_CLIENTE,
    CASE WHEN UNIFORM(0, 7, RANDOM()) < 4 
        THEN ARRAY_CONSTRUCT('Juan', 'María', 'Carlos', 'Ana', 'Pedro', 'Laura')[UNIFORM(0, 5, RANDOM())]::VARCHAR || ' ' ||
             ARRAY_CONSTRUCT('García', 'López', 'Martínez', 'Rodríguez', 'Hernández')[UNIFORM(0, 4, RANDOM())]::VARCHAR
        ELSE ARRAY_CONSTRUCT('Empresa', 'Comercial', 'Industrial', 'Servicios')[UNIFORM(0, 3, RANDOM())]::VARCHAR || ' ' ||
             ARRAY_CONSTRUCT('Alpha', 'Beta', 'Gamma', 'Delta')[UNIFORM(0, 3, RANDOM())]::VARCHAR || ' SA'
    END AS NOMBRE,
    'Calle ' || UNIFORM(1, 500, RANDOM()) || ' #' || UNIFORM(1, 999, RANDOM()) AS DIRECCION,
    ARRAY_CONSTRUCT('Ciudad de México', 'Monterrey', 'Guadalajara', 'Puebla', 'Querétaro', 'Mérida', 'León', 'Tijuana')[UNIFORM(0, 7, RANDOM())]::VARCHAR AS CIUDAD,
    LPAD(UNIFORM(10000, 99999, RANDOM())::VARCHAR, 5, '0') AS CODIGO_POSTAL,
    CASE UNIFORM(0, 7, RANDOM())
        WHEN 0 THEN 'DAC' -- Doméstico Alto Consumo
        WHEN 1 THEN '1' -- Residencial básico
        WHEN 2 THEN '1A'
        WHEN 3 THEN '1B'
        WHEN 4 THEN 'PDBT' -- Pequeña demanda baja tensión
        WHEN 5 THEN 'GDBT' -- Gran demanda baja tensión
        ELSE 'GDMT' -- Gran demanda media tensión
    END AS TARIFA,
    DATEADD('day', -UNIFORM(30, 3650, RANDOM()), CURRENT_DATE()) AS FECHA_ALTA,
    'MED' || LPAD(SEQ4()::VARCHAR, 10, '0') AS MEDIDOR_ID,
    CASE WHEN UNIFORM(1, 10, RANDOM()) <= 7 THEN 'SMART' ELSE 'TRADICIONAL' END AS TIPO_MEDIDOR,
    ARRAY_CONSTRUCT('ACTIVO', 'ACTIVO', 'ACTIVO', 'ACTIVO', 'SUSPENDIDO', 'BAJA')[UNIFORM(0, 5, RANDOM())]::VARCHAR AS ESTATUS
FROM TABLE(GENERATOR(ROWCOUNT => 10000));
```

### Tabla: LECTURAS_MEDIDOR
```sql
CREATE OR REPLACE TABLE RAW.LECTURAS (
    ID_LECTURA VARCHAR(20) PRIMARY KEY,
    ID_CLIENTE VARCHAR(15),
    MEDIDOR_ID VARCHAR(20),
    FECHA_HORA TIMESTAMP,
    LECTURA_KWH NUMBER(12,2),
    CONSUMO_PERIODO NUMBER(10,2),
    DEMANDA_KW NUMBER(8,2),
    FACTOR_POTENCIA NUMBER(4,2),
    VOLTAJE NUMBER(6,2),
    TIPO_LECTURA VARCHAR(20), -- REAL, ESTIMADA, AJUSTADA
    FOREIGN KEY (ID_CLIENTE) REFERENCES RAW.CLIENTES(ID_CLIENTE)
);

-- Generar lecturas (hourly para smart meters)
INSERT INTO RAW.LECTURAS
SELECT 
    'LEC' || TO_CHAR(CURRENT_DATE(), 'YYYYMMDD') || LPAD(SEQ4()::VARCHAR, 10, '0') AS ID_LECTURA,
    'CLI' || LPAD(UNIFORM(1, 10000, RANDOM())::VARCHAR, 8, '0') AS ID_CLIENTE,
    'MED' || LPAD(UNIFORM(1, 10000, RANDOM())::VARCHAR, 10, '0') AS MEDIDOR_ID,
    DATEADD('hour', -UNIFORM(1, 8760, RANDOM()), CURRENT_TIMESTAMP()) AS FECHA_HORA,
    ROUND(UNIFORM(100, 100000, RANDOM())::FLOAT, 2) AS LECTURA_KWH,
    ROUND(UNIFORM(1, 500, RANDOM())::FLOAT, 2) AS CONSUMO_PERIODO,
    ROUND(UNIFORM(1, 200, RANDOM())::FLOAT, 2) AS DEMANDA_KW,
    ROUND(UNIFORM(85, 100, RANDOM())::FLOAT / 100, 2) AS FACTOR_POTENCIA,
    ROUND(UNIFORM(118, 122, RANDOM())::FLOAT, 2) AS VOLTAJE,
    ARRAY_CONSTRUCT('REAL', 'REAL', 'REAL', 'REAL', 'ESTIMADA')[UNIFORM(0, 4, RANDOM())]::VARCHAR AS TIPO_LECTURA
FROM TABLE(GENERATOR(ROWCOUNT => 500000));
```

### Tabla: FACTURAS
```sql
CREATE OR REPLACE TABLE RAW.FACTURAS (
    ID_FACTURA VARCHAR(15) PRIMARY KEY,
    ID_CLIENTE VARCHAR(15),
    PERIODO_INICIO DATE,
    PERIODO_FIN DATE,
    FECHA_EMISION DATE,
    FECHA_VENCIMIENTO DATE,
    CONSUMO_KWH NUMBER(10,2),
    DEMANDA_MAXIMA_KW NUMBER(8,2),
    IMPORTE_ENERGIA NUMBER(12,2),
    IMPORTE_DEMANDA NUMBER(12,2),
    IMPORTE_TRANSMISION NUMBER(12,2),
    IMPORTE_DISTRIBUCION NUMBER(12,2),
    SUBSIDIO NUMBER(12,2),
    IVA NUMBER(12,2),
    TOTAL NUMBER(12,2),
    ESTATUS VARCHAR(20), -- EMITIDA, PAGADA, VENCIDA, CANCELADA
    FECHA_PAGO DATE,
    FOREIGN KEY (ID_CLIENTE) REFERENCES RAW.CLIENTES(ID_CLIENTE)
);

-- Generar facturas
INSERT INTO RAW.FACTURAS
SELECT 
    'FAC' || LPAD(SEQ4()::VARCHAR, 10, '0') AS ID_FACTURA,
    'CLI' || LPAD(UNIFORM(1, 10000, RANDOM())::VARCHAR, 8, '0') AS ID_CLIENTE,
    DATEADD('month', -UNIFORM(1, 24, RANDOM()), DATE_TRUNC('MONTH', CURRENT_DATE())) AS PERIODO_INICIO,
    DATEADD('day', -1, DATEADD('month', 1, DATEADD('month', -UNIFORM(1, 24, RANDOM()), DATE_TRUNC('MONTH', CURRENT_DATE())))) AS PERIODO_FIN,
    DATEADD('day', 5, DATEADD('month', -UNIFORM(1, 24, RANDOM()), DATE_TRUNC('MONTH', CURRENT_DATE()))) AS FECHA_EMISION,
    DATEADD('day', 20, DATEADD('month', -UNIFORM(1, 24, RANDOM()), DATE_TRUNC('MONTH', CURRENT_DATE()))) AS FECHA_VENCIMIENTO,
    ROUND(UNIFORM(50, 5000, RANDOM())::FLOAT, 2) AS CONSUMO_KWH,
    ROUND(UNIFORM(1, 100, RANDOM())::FLOAT, 2) AS DEMANDA_MAXIMA_KW,
    ROUND(UNIFORM(100, 5000, RANDOM())::FLOAT, 2) AS IMPORTE_ENERGIA,
    ROUND(UNIFORM(50, 500, RANDOM())::FLOAT, 2) AS IMPORTE_DEMANDA,
    ROUND(UNIFORM(20, 200, RANDOM())::FLOAT, 2) AS IMPORTE_TRANSMISION,
    ROUND(UNIFORM(30, 300, RANDOM())::FLOAT, 2) AS IMPORTE_DISTRIBUCION,
    ROUND(UNIFORM(0, 500, RANDOM())::FLOAT, 2) AS SUBSIDIO,
    ROUND(UNIFORM(50, 800, RANDOM())::FLOAT, 2) AS IVA,
    ROUND(UNIFORM(200, 8000, RANDOM())::FLOAT, 2) AS TOTAL,
    ARRAY_CONSTRUCT('PAGADA', 'PAGADA', 'PAGADA', 'PAGADA', 'EMITIDA', 'VENCIDA')[UNIFORM(0, 5, RANDOM())]::VARCHAR AS ESTATUS,
    CASE WHEN UNIFORM(0, 5, RANDOM()) < 4 
         THEN DATEADD('day', UNIFORM(1, 25, RANDOM()), DATEADD('month', -UNIFORM(1, 24, RANDOM()), DATE_TRUNC('MONTH', CURRENT_DATE())))
         ELSE NULL END AS FECHA_PAGO
FROM TABLE(GENERATOR(ROWCOUNT => 50000));
```

CREATE OR REPLACE TABLE RAW.ALERTAS (
    ID_ALERTA VARCHAR(15) PRIMARY KEY,
    ID_CLIENTE VARCHAR(15),
    FECHA_GENERACION TIMESTAMP,
    TIPO_ALERTA VARCHAR(50),
    DESCRIPCION VARCHAR(300),
    VALOR_DETECTADO NUMBER(12,2),
    VALOR_ESPERADO NUMBER(12,2),
    DESVIACION_PCT NUMBER(6,2),
    PRIORIDAD VARCHAR(10),
    ESTATUS VARCHAR(20),
    FECHA_CIERRE TIMESTAMP,
    FOREIGN KEY (ID_CLIENTE) REFERENCES RAW.CLIENTES(ID_CLIENTE)
);

-- Generar alertas
INSERT INTO RAW.ALERTAS
SELECT 
    'ALR' || LPAD(SEQ4()::VARCHAR, 8, '0') AS ID_ALERTA,
    'CLI' || LPAD(UNIFORM(1, 10000, RANDOM())::VARCHAR, 8, '0') AS ID_CLIENTE,
    DATEADD('hour', -UNIFORM(1, 2160, RANDOM()), CURRENT_TIMESTAMP()) AS FECHA_GENERACION,
    ARRAY_CONSTRUCT(
        'Consumo excesivo',
        'Consumo bajo inusual',
        'Posible fraude/bypass',
        'Falla de medidor',
        'Voltaje fuera de rango',
        'Factor de potencia bajo'
    )[UNIFORM(0, 5, RANDOM())]::VARCHAR AS TIPO_ALERTA,
    'Anomalía detectada en el consumo del cliente' AS DESCRIPCION,
    ROUND(UNIFORM(100, 10000, RANDOM())::FLOAT, 2) AS VALOR_DETECTADO,
    ROUND(UNIFORM(50, 5000, RANDOM())::FLOAT, 2) AS VALOR_ESPERADO,
    ROUND(UNIFORM(-50, 200, RANDOM())::FLOAT, 2) AS DESVIACION_PCT,
    ARRAY_CONSTRUCT('ALTA', 'MEDIA', 'MEDIA', 'BAJA')[UNIFORM(0, 3, RANDOM())]::VARCHAR AS PRIORIDAD,
    ARRAY_CONSTRUCT('ABIERTA', 'EN_REVISION', 'CERRADA', 'FALSO_POSITIVO')[UNIFORM(0, 3, RANDOM())]::VARCHAR AS ESTATUS,
    CASE WHEN UNIFORM(1, 10, RANDOM()) >= 5 
         THEN DATEADD('hour', -UNIFORM(1, 720, RANDOM()), CURRENT_TIMESTAMP()) 
         ELSE NULL END AS FECHA_CIERRE
FROM TABLE(GENERATOR(ROWCOUNT => 2000));

## Vistas Analíticas

```sql
-- Vista de análisis de consumo
CREATE OR REPLACE VIEW ANALYTICS.V_ANALISIS_CONSUMO AS
SELECT 
    DATE_TRUNC('MONTH', l.FECHA_HORA) AS MES,
    c.TIPO_CLIENTE,
    c.TARIFA,
    c.CIUDAD,
    
    -- Volumen
    COUNT(DISTINCT c.ID_CLIENTE) AS CLIENTES,
    SUM(l.CONSUMO_PERIODO) AS CONSUMO_TOTAL_KWH,
    ROUND(AVG(l.CONSUMO_PERIODO), 2) AS CONSUMO_PROMEDIO_KWH,
    
    -- Demanda
    MAX(l.DEMANDA_KW) AS DEMANDA_MAXIMA_KW,
    ROUND(AVG(l.DEMANDA_KW), 2) AS DEMANDA_PROMEDIO_KW,
    
    -- Calidad
    ROUND(AVG(l.FACTOR_POTENCIA), 3) AS FP_PROMEDIO,
    ROUND(AVG(l.VOLTAJE), 2) AS VOLTAJE_PROMEDIO

FROM RAW.LECTURAS l
JOIN RAW.CLIENTES c ON l.ID_CLIENTE = c.ID_CLIENTE
GROUP BY DATE_TRUNC('MONTH', l.FECHA_HORA), c.TIPO_CLIENTE, c.TARIFA, c.CIUDAD;

-- Vista de facturación y cartera
CREATE OR REPLACE VIEW ANALYTICS.V_FACTURACION_CARTERA AS
SELECT 
    DATE_TRUNC('MONTH', f.FECHA_EMISION) AS MES,
    c.TIPO_CLIENTE,
    c.CIUDAD,
    
    -- Facturación
    COUNT(f.ID_FACTURA) AS FACTURAS_EMITIDAS,
    SUM(f.TOTAL) AS FACTURACION_TOTAL,
    ROUND(AVG(f.TOTAL), 2) AS FACTURA_PROMEDIO,
    
    -- Consumo facturado
    SUM(f.CONSUMO_KWH) AS CONSUMO_TOTAL_KWH,
    
    -- Cartera
    SUM(CASE WHEN f.ESTATUS = 'PAGADA' THEN f.TOTAL ELSE 0 END) AS RECAUDACION,
    SUM(CASE WHEN f.ESTATUS = 'VENCIDA' THEN f.TOTAL ELSE 0 END) AS CARTERA_VENCIDA,
    ROUND(SUM(CASE WHEN f.ESTATUS = 'VENCIDA' THEN f.TOTAL ELSE 0 END) / NULLIF(SUM(f.TOTAL), 0) * 100, 2) AS PCT_CARTERA_VENCIDA,
    
    -- Subsidio
    SUM(f.SUBSIDIO) AS SUBSIDIO_TOTAL

FROM RAW.FACTURAS f
JOIN RAW.CLIENTES c ON f.ID_CLIENTE = c.ID_CLIENTE
GROUP BY DATE_TRUNC('MONTH', f.FECHA_EMISION), c.TIPO_CLIENTE, c.CIUDAD;

-- Vista de alertas y anomalías
CREATE OR REPLACE VIEW ANALYTICS.V_ALERTAS_ANOMALIAS AS
SELECT 
    DATE_TRUNC('WEEK', a.FECHA_GENERACION) AS SEMANA,
    a.TIPO_ALERTA,
    a.PRIORIDAD,
    c.TIPO_CLIENTE,
    
    COUNT(*) AS TOTAL_ALERTAS,
    COUNT(CASE WHEN a.ESTATUS = 'ABIERTA' THEN 1 END) AS ALERTAS_ABIERTAS,
    COUNT(CASE WHEN a.ESTATUS = 'FALSO_POSITIVO' THEN 1 END) AS FALSOS_POSITIVOS,
    ROUND(SUM(a.VALOR_DETECTADO - a.VALOR_ESPERADO), 2) AS PERDIDA_POTENCIAL,
    ROUND(AVG(a.DESVIACION_PCT), 2) AS DESVIACION_PROMEDIO

FROM RAW.ALERTAS a
JOIN RAW.CLIENTES c ON a.ID_CLIENTE = c.ID_CLIENTE
GROUP BY DATE_TRUNC('WEEK', a.FECHA_GENERACION), a.TIPO_ALERTA, a.PRIORIDAD, c.TIPO_CLIENTE;

-- Vista de curva de carga
CREATE OR REPLACE VIEW ANALYTICS.V_CURVA_CARGA AS
SELECT 
    HOUR(l.FECHA_HORA) AS HORA,
    DAYNAME(l.FECHA_HORA) AS DIA_SEMANA,
    c.TIPO_CLIENTE,
    
    ROUND(AVG(l.DEMANDA_KW), 2) AS DEMANDA_PROMEDIO,
    MAX(l.DEMANDA_KW) AS DEMANDA_MAXIMA,
    MIN(l.DEMANDA_KW) AS DEMANDA_MINIMA

FROM RAW.LECTURAS l
JOIN RAW.CLIENTES c ON l.ID_CLIENTE = c.ID_CLIENTE
WHERE l.FECHA_HORA >= DATEADD('day', -30, CURRENT_TIMESTAMP())
GROUP BY HOUR(l.FECHA_HORA), DAYNAME(l.FECHA_HORA), c.TIPO_CLIENTE;
```
