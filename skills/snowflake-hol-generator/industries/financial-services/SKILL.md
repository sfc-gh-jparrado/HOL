### Tabla: CLIENTES_FINANCIEROS
```sql
CREATE OR REPLACE TABLE RAW.CLIENTES (
    ID_CLIENTE VARCHAR(15) PRIMARY KEY,
    TIPO_CLIENTE VARCHAR(20), -- PERSONA_FISICA, PERSONA_MORAL
    NOMBRE VARCHAR(200),
    RFC VARCHAR(13),
    FECHA_ALTA DATE,
    SEGMENTO VARCHAR(30), -- PREMIUM, PREFERENTE, MASIVO
    SCORE_CREDITICIO NUMBER(3),
    NIVEL_RIESGO VARCHAR(10), -- BAJO, MEDIO, ALTO
    EJECUTIVO_ASIGNADO VARCHAR(100),
    SUCURSAL_ORIGEN VARCHAR(50),
    CANAL_CAPTACION VARCHAR(30),
    ESTATUS VARCHAR(20) DEFAULT 'ACTIVO'
);

-- Generar clientes
INSERT INTO RAW.CLIENTES
SELECT 
    'CLI' || LPAD(SEQ4()::VARCHAR, 8, '0') AS ID_CLIENTE,
    CASE WHEN UNIFORM(1, 10, RANDOM()) <= 8 THEN 'PERSONA_FISICA' ELSE 'PERSONA_MORAL' END AS TIPO_CLIENTE,
    CASE WHEN UNIFORM(1, 10, RANDOM()) <= 8 
        THEN ARRAY_CONSTRUCT('Juan', 'María', 'Carlos', 'Ana', 'Pedro', 'Laura', 'Roberto', 'Elena')[UNIFORM(0, 7, RANDOM())]::VARCHAR || ' ' ||
             ARRAY_CONSTRUCT('García', 'López', 'Martínez', 'Rodríguez', 'Hernández', 'González')[UNIFORM(0, 5, RANDOM())]::VARCHAR || ' ' ||
             ARRAY_CONSTRUCT('Pérez', 'Sánchez', 'Ramírez', 'Torres', 'Flores')[UNIFORM(0, 4, RANDOM())]::VARCHAR
        ELSE ARRAY_CONSTRUCT('Corporativo', 'Grupo', 'Servicios', 'Comercial', 'Industrial')[UNIFORM(0, 4, RANDOM())]::VARCHAR || ' ' ||
             ARRAY_CONSTRUCT('Alpha', 'Beta', 'Delta', 'Sigma', 'Omega')[UNIFORM(0, 4, RANDOM())]::VARCHAR || ' SA de CV'
    END AS NOMBRE,
    UPPER(SUBSTRING(MD5(RANDOM()::VARCHAR), 1, 13)) AS RFC,
    DATEADD('day', -UNIFORM(30, 3650, RANDOM()), CURRENT_DATE()) AS FECHA_ALTA,
    ARRAY_CONSTRUCT('PREMIUM', 'PREMIUM', 'PREFERENTE', 'PREFERENTE', 'PREFERENTE', 'MASIVO', 'MASIVO', 'MASIVO', 'MASIVO', 'MASIVO')[UNIFORM(0, 9, RANDOM())]::VARCHAR AS SEGMENTO,
    UNIFORM(300, 850, RANDOM()) AS SCORE_CREDITICIO,
    CASE 
        WHEN UNIFORM(300, 850, RANDOM()) >= 700 THEN 'BAJO'
        WHEN UNIFORM(300, 850, RANDOM()) >= 550 THEN 'MEDIO'
        ELSE 'ALTO'
    END AS NIVEL_RIESGO,
    'Ejecutivo ' || UNIFORM(1, 50, RANDOM()) AS EJECUTIVO_ASIGNADO,
    ARRAY_CONSTRUCT('Centro', 'Norte', 'Sur', 'Oriente', 'Poniente', 'Digital')[UNIFORM(0, 5, RANDOM())]::VARCHAR AS SUCURSAL_ORIGEN,
    ARRAY_CONSTRUCT('Sucursal', 'Digital', 'Referido', 'Campaña', 'Alianza')[UNIFORM(0, 4, RANDOM())]::VARCHAR AS CANAL_CAPTACION,
    'ACTIVO' AS ESTATUS
FROM TABLE(GENERATOR(ROWCOUNT => 5000));
```

### Tabla: CUENTAS
```sql
CREATE OR REPLACE TABLE RAW.CUENTAS (
    ID_CUENTA VARCHAR(20) PRIMARY KEY,
    ID_CLIENTE VARCHAR(15),
    TIPO_CUENTA VARCHAR(30), -- AHORRO, CHEQUES, INVERSION, CREDITO
    NUMERO_CUENTA VARCHAR(20),
    FECHA_APERTURA DATE,
    SALDO_ACTUAL NUMBER(15,2),
    SALDO_PROMEDIO NUMBER(15,2),
    MONEDA VARCHAR(3) DEFAULT 'MXN',
    TASA_INTERES NUMBER(5,2),
    ESTATUS VARCHAR(20),
    FOREIGN KEY (ID_CLIENTE) REFERENCES RAW.CLIENTES(ID_CLIENTE)
);

-- Generar cuentas
INSERT INTO RAW.CUENTAS
SELECT 
    'CTA' || LPAD(SEQ4()::VARCHAR, 10, '0') AS ID_CUENTA,
    'CLI' || LPAD(UNIFORM(1, 5000, RANDOM())::VARCHAR, 8, '0') AS ID_CLIENTE,
    ARRAY_CONSTRUCT('AHORRO', 'AHORRO', 'CHEQUES', 'INVERSION', 'CREDITO')[UNIFORM(0, 4, RANDOM())]::VARCHAR AS TIPO_CUENTA,
    LPAD(UNIFORM(1000000000, 9999999999, RANDOM())::VARCHAR, 10, '0') AS NUMERO_CUENTA,
    DATEADD('day', -UNIFORM(1, 1825, RANDOM()), CURRENT_DATE()) AS FECHA_APERTURA,
    ROUND(UNIFORM(0, 5000000, RANDOM())::FLOAT, 2) AS SALDO_ACTUAL,
    ROUND(UNIFORM(0, 4000000, RANDOM())::FLOAT, 2) AS SALDO_PROMEDIO,
    CASE WHEN UNIFORM(1, 10, RANDOM()) <= 9 THEN 'MXN' ELSE 'USD' END AS MONEDA,
    ROUND(UNIFORM(0, 15, RANDOM())::FLOAT, 2) AS TASA_INTERES,
    ARRAY_CONSTRUCT('ACTIVA', 'ACTIVA', 'ACTIVA', 'ACTIVA', 'INACTIVA', 'BLOQUEADA')[UNIFORM(0, 5, RANDOM())]::VARCHAR AS ESTATUS
FROM TABLE(GENERATOR(ROWCOUNT => 8000));
```

CREATE OR REPLACE TABLE RAW.CREDITOS (

    MONTO_ORIGINAL NUMBER(15,2),
    SALDO_ACTUAL NUMBER(15,2),
    TASA_ANUAL NUMBER(5,2),
    PLAZO_MESES NUMBER,
    FECHA_OTORGAMIENTO DATE,
    FECHA_VENCIMIENTO DATE,
    DIA_CORTE NUMBER(2),
    ESTATUS VARCHAR(20), -- AL_CORRIENTE, ATRASO_30, ATRASO_60, ATRASO_90, VENCIDO, LIQUIDADO
    DIAS_ATRASO NUMBER DEFAULT 0,
    FOREIGN KEY (ID_CLIENTE) REFERENCES RAW.CLIENTES(ID_CLIENTE)
);

-- Generar créditos
INSERT INTO RAW.CREDITOS
SELECT 
    'CRE' || LPAD(SEQ4()::VARCHAR, 8, '0') AS ID_CREDITO,
    'CLI' || LPAD(UNIFORM(1, 5000, RANDOM())::VARCHAR, 8, '0') AS ID_CLIENTE,
    ARRAY_CONSTRUCT('PERSONAL', 'PERSONAL', 'HIPOTECARIO', 'AUTOMOTRIZ', 'TARJETA', 'TARJETA', 'EMPRESARIAL')[UNIFORM(0, 6, RANDOM())]::VARCHAR AS TIPO_CREDITO,
    ROUND(UNIFORM(10000, 5000000, RANDOM())::FLOAT, 2) AS MONTO_ORIGINAL,
    ROUND(UNIFORM(5000, 4000000, RANDOM())::FLOAT, 2) AS SALDO_ACTUAL,
    ROUND(UNIFORM(8, 45, RANDOM())::FLOAT, 2) AS TASA_ANUAL,
    ARRAY_CONSTRUCT(12, 24, 36, 48, 60, 120, 180, 240)[UNIFORM(0, 7, RANDOM())] AS PLAZO_MESES,
    DATEADD('day', -UNIFORM(30, 1825, RANDOM()), CURRENT_DATE()) AS FECHA_OTORGAMIENTO,
    DATEADD('month', UNIFORM(12, 240, RANDOM()), CURRENT_DATE()) AS FECHA_VENCIMIENTO,
    UNIFORM(1, 28, RANDOM()) AS DIA_CORTE,
    ARRAY_CONSTRUCT('AL_CORRIENTE', 'AL_CORRIENTE', 'AL_CORRIENTE', 'AL_CORRIENTE', 'ATRASO_30', 'ATRASO_60', 'ATRASO_90', 'LIQUIDADO')[UNIFORM(0, 7, RANDOM())]::VARCHAR AS ESTATUS,
    CASE 
        WHEN UNIFORM(1, 10, RANDOM()) <= 7 THEN 0
        ELSE UNIFORM(1, 180, RANDOM())
    END AS DIAS_ATRASO
FROM TABLE(GENERATOR(ROWCOUNT => 3000));

## Vistas Analíticas

```sql
-- Vista de cartera de clientes
CREATE OR REPLACE VIEW ANALYTICS.V_CARTERA_CLIENTES AS
SELECT 
    c.ID_CLIENTE,
    c.NOMBRE,
    c.TIPO_CLIENTE,
    c.SEGMENTO,
    c.SCORE_CREDITICIO,
    c.NIVEL_RIESGO,
    c.SUCURSAL_ORIGEN,
    
    -- Cuentas
    COUNT(DISTINCT ct.ID_CUENTA) AS NUM_CUENTAS,
    SUM(ct.SALDO_ACTUAL) AS SALDO_TOTAL,
    
    -- Créditos
    COUNT(DISTINCT cr.ID_CREDITO) AS NUM_CREDITOS,
    SUM(cr.SALDO_ACTUAL) AS DEUDA_TOTAL,
    MAX(cr.DIAS_ATRASO) AS MAX_DIAS_ATRASO,
    
    -- Transaccionalidad (últimos 30 días)
    (SELECT COUNT(*) FROM RAW.TRANSACCIONES t 
     WHERE t.ID_CUENTA IN (SELECT ID_CUENTA FROM RAW.CUENTAS WHERE ID_CLIENTE = c.ID_CLIENTE)
     AND t.FECHA_HORA >= DATEADD('day', -30, CURRENT_TIMESTAMP())) AS TXN_30_DIAS,
    
    -- Clasificación de valor
    CASE 
        WHEN SUM(ct.SALDO_ACTUAL) > 1000000 THEN '💎 Alto Valor'
        WHEN SUM(ct.SALDO_ACTUAL) > 250000 THEN '🥇 Valor Medio'
        ELSE '🥉 Valor Base'
    END AS CLASIFICACION_VALOR

FROM RAW.CLIENTES c
LEFT JOIN RAW.CUENTAS ct ON c.ID_CLIENTE = ct.ID_CLIENTE
LEFT JOIN RAW.CREDITOS cr ON c.ID_CLIENTE = cr.ID_CLIENTE
WHERE c.ESTATUS = 'ACTIVO'
GROUP BY c.ID_CLIENTE, c.NOMBRE, c.TIPO_CLIENTE, c.SEGMENTO, c.SCORE_CREDITICIO, c.NIVEL_RIESGO, c.SUCURSAL_ORIGEN;

-- Vista de cartera de crédito
CREATE OR REPLACE VIEW ANALYTICS.V_CARTERA_CREDITO AS
SELECT 
    cr.TIPO_CREDITO,
    cr.ESTATUS,
    COUNT(*) AS NUM_CREDITOS,
    SUM(cr.MONTO_ORIGINAL) AS MONTO_ORIGINADO,
    SUM(cr.SALDO_ACTUAL) AS SALDO_CARTERA,
    ROUND(AVG(cr.TASA_ANUAL), 2) AS TASA_PROMEDIO,
    SUM(CASE WHEN cr.DIAS_ATRASO > 0 THEN cr.SALDO_ACTUAL ELSE 0 END) AS CARTERA_VENCIDA,
    ROUND(SUM(CASE WHEN cr.DIAS_ATRASO > 0 THEN cr.SALDO_ACTUAL ELSE 0 END) / NULLIF(SUM(cr.SALDO_ACTUAL), 0) * 100, 2) AS PCT_CARTERA_VENCIDA
FROM RAW.CREDITOS cr
GROUP BY cr.TIPO_CREDITO, cr.ESTATUS;

-- Vista de métricas AML
CREATE OR REPLACE VIEW ANALYTICS.V_METRICAS_AML AS
SELECT 
    DATE_TRUNC('MONTH', a.FECHA_GENERACION) AS MES,
    a.TIPO_ALERTA,
    a.NIVEL_RIESGO,
    COUNT(*) AS TOTAL_ALERTAS,
    SUM(a.MONTO_INVOLUCRADO) AS MONTO_TOTAL_ALERTAS,
    COUNT(CASE WHEN a.ESTATUS = 'CERRADA_POSITIVO' THEN 1 END) AS ALERTAS_CONFIRMADAS,
    COUNT(CASE WHEN a.ESTATUS = 'CERRADA_FALSO_POSITIVO' THEN 1 END) AS FALSOS_POSITIVOS,
    ROUND(COUNT(CASE WHEN a.ESTATUS = 'CERRADA_POSITIVO' THEN 1 END)::FLOAT / 
          NULLIF(COUNT(CASE WHEN a.ESTATUS LIKE 'CERRADA%' THEN 1 END), 0) * 100, 2) AS TASA_VERDADEROS_POSITIVOS
FROM RAW.ALERTAS_AML a
GROUP BY DATE_TRUNC('MONTH', a.FECHA_GENERACION), a.TIPO_ALERTA, a.NIVEL_RIESGO;
```

## Análisis con Cortex AI

```sql
-- Análisis de riesgo con IA
CREATE OR REPLACE VIEW ANALYTICS.V_RIESGO_IA AS
SELECT 
    c.ID_CLIENTE,
    c.NOMBRE,
    c.SCORE_CREDITICIO,
    cr.SALDO_ACTUAL AS DEUDA,
    cr.DIAS_ATRASO,
    
    -- Clasificación de riesgo con IA
    SNOWFLAKE.CORTEX.CLASSIFY_TEXT(
        'Cliente con score ' || c.SCORE_CREDITICIO || ', deuda de ' || cr.SALDO_ACTUAL || 
        ' pesos y ' || cr.DIAS_ATRASO || ' días de atraso en pagos',
        ['Riesgo Bajo - Cliente confiable', 'Riesgo Medio - Monitorear', 'Riesgo Alto - Acción inmediata']
    ):label::STRING AS CLASIFICACION_IA,
    
    -- Recomendación de acción
    SNOWFLAKE.CORTEX.COMPLETE(
        'mistral-large',
        'Eres un analista de riesgo crediticio. Basado en estos datos del cliente: ' ||
        'Score crediticio: ' || c.SCORE_CREDITICIO || ', ' ||
        'Días de atraso: ' || cr.DIAS_ATRASO || ', ' ||
        'Saldo pendiente: $' || cr.SALDO_ACTUAL || '. ' ||
        'Recomienda UNA acción específica en máximo 15 palabras.'
    ) AS RECOMENDACION_IA

FROM RAW.CLIENTES c
JOIN RAW.CREDITOS cr ON c.ID_CLIENTE = cr.ID_CLIENTE
WHERE cr.ESTATUS != 'LIQUIDADO';
```
