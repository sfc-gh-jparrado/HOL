-- Generar historial de suscripciones
INSERT INTO RAW.SUSCRIPCIONES
SELECT 
    'SUB' || LPAD(SEQ4()::VARCHAR, 8, '0') AS ID_SUSCRIPCION,
    'ACC' || LPAD(UNIFORM(1, 2000, RANDOM())::VARCHAR, 6, '0') AS ID_CUENTA,

    CASE UNIFORM(1, 5, RANDOM())
        WHEN 1 THEN 0
        WHEN 2 THEN 29
        WHEN 3 THEN 99
        WHEN 4 THEN 299
        ELSE UNIFORM(500, 5000, RANDOM())
    END AS MRR,
    CASE UNIFORM(1, 5, RANDOM())
        WHEN 1 THEN 0
        WHEN 2 THEN 290
        WHEN 3 THEN 990
        WHEN 4 THEN 2990
        ELSE UNIFORM(6000, 60000, RANDOM())
    END AS ARR,
    ARRAY_CONSTRUCT('NEW', 'NEW', 'UPGRADE', 'DOWNGRADE', 'CHURN', 'REACTIVATION')[UNIFORM(0, 5, RANDOM())]::VARCHAR AS TIPO_CAMBIO,

## Vistas Analíticas

```sql
-- Vista de métricas SaaS
CREATE OR REPLACE VIEW ANALYTICS.V_METRICAS_SAAS AS
WITH METRICAS_MES AS (
    SELECT 
        DATE_TRUNC('MONTH', s.FECHA_INICIO) AS MES,
        SUM(CASE WHEN s.TIPO_CAMBIO = 'NEW' THEN s.MRR ELSE 0 END) AS NEW_MRR,
        SUM(CASE WHEN s.TIPO_CAMBIO = 'UPGRADE' THEN s.MRR ELSE 0 END) AS EXPANSION_MRR,
        SUM(CASE WHEN s.TIPO_CAMBIO = 'DOWNGRADE' THEN s.MRR ELSE 0 END) AS CONTRACTION_MRR,
        SUM(CASE WHEN s.TIPO_CAMBIO = 'CHURN' THEN s.MRR ELSE 0 END) AS CHURNED_MRR,
        SUM(CASE WHEN s.TIPO_CAMBIO = 'REACTIVATION' THEN s.MRR ELSE 0 END) AS REACTIVATION_MRR,
        COUNT(CASE WHEN s.TIPO_CAMBIO = 'NEW' THEN 1 END) AS NEW_CUSTOMERS,
        COUNT(CASE WHEN s.TIPO_CAMBIO = 'CHURN' THEN 1 END) AS CHURNED_CUSTOMERS
    FROM RAW.SUSCRIPCIONES s
    GROUP BY DATE_TRUNC('MONTH', s.FECHA_INICIO)
)
SELECT 
    MES,
    NEW_MRR,
    EXPANSION_MRR,
    CONTRACTION_MRR,
    CHURNED_MRR,
    REACTIVATION_MRR,
    NEW_MRR + EXPANSION_MRR + REACTIVATION_MRR - CONTRACTION_MRR - CHURNED_MRR AS NET_NEW_MRR,
    NEW_CUSTOMERS,
    CHURNED_CUSTOMERS,
    ROUND(CHURNED_CUSTOMERS::FLOAT / NULLIF(LAG(NEW_CUSTOMERS) OVER (ORDER BY MES) + NEW_CUSTOMERS, 0) * 100, 2) AS CHURN_RATE_PCT
FROM METRICAS_MES
ORDER BY MES DESC;

-- Vista de engagement por cuenta
CREATE OR REPLACE VIEW ANALYTICS.V_ENGAGEMENT_CUENTAS AS
SELECT 
    a.ID_CUENTA,
    a.NOMBRE_EMPRESA,
    a.TAMANO_EMPRESA,
    a.MRR,
    a.HEALTH_SCORE,
    a.ESTATUS_SUSCRIPCION,
    p.NOMBRE_PLAN,
    
    -- Usuarios
    COUNT(DISTINCT u.ID_USUARIO) AS TOTAL_USUARIOS,
    COUNT(DISTINCT CASE WHEN u.ULTIMO_LOGIN >= DATEADD('day', -7, CURRENT_TIMESTAMP()) THEN u.ID_USUARIO END) AS USUARIOS_ACTIVOS_7D,
    
    -- Eventos (últimos 30 días)
    COUNT(e.ID_EVENTO) AS EVENTOS_30D,
    COUNT(DISTINCT e.FEATURE) AS FEATURES_USADAS,
    
    -- Clasificación de riesgo
    CASE 
        WHEN a.HEALTH_SCORE >= 80 AND COUNT(DISTINCT CASE WHEN u.ULTIMO_LOGIN >= DATEADD('day', -7, CURRENT_TIMESTAMP()) THEN u.ID_USUARIO END) > 0 THEN '🟢 Saludable'
        WHEN a.HEALTH_SCORE >= 50 THEN '🟡 Atención'
        ELSE '🔴 Riesgo de Churn'
    END AS ESTADO_CUENTA

FROM RAW.CUENTAS a
JOIN RAW.PLANES p ON a.ID_PLAN = p.ID_PLAN
LEFT JOIN RAW.USUARIOS u ON a.ID_CUENTA = u.ID_CUENTA
LEFT JOIN RAW.EVENTOS e ON a.ID_CUENTA = e.ID_CUENTA 
    AND e.TIMESTAMP_EVENTO >= DATEADD('day', -30, CURRENT_TIMESTAMP())
WHERE a.ESTATUS_SUSCRIPCION = 'ACTIVE'
GROUP BY a.ID_CUENTA, a.NOMBRE_EMPRESA, a.TAMANO_EMPRESA, a.MRR, a.HEALTH_SCORE, a.ESTATUS_SUSCRIPCION, p.NOMBRE_PLAN;

-- Vista de cohortes
CREATE OR REPLACE VIEW ANALYTICS.V_COHORTES AS
SELECT 
    DATE_TRUNC('MONTH', a.FECHA_REGISTRO) AS COHORTE,
    COUNT(DISTINCT a.ID_CUENTA) AS CUENTAS_INICIALES,
    COUNT(DISTINCT CASE WHEN a.ESTATUS_SUSCRIPCION = 'ACTIVE' THEN a.ID_CUENTA END) AS CUENTAS_ACTIVAS,
    ROUND(COUNT(DISTINCT CASE WHEN a.ESTATUS_SUSCRIPCION = 'ACTIVE' THEN a.ID_CUENTA END)::FLOAT / 
          NULLIF(COUNT(DISTINCT a.ID_CUENTA), 0) * 100, 2) AS RETENCION_PCT,
    SUM(a.MRR) AS MRR_COHORTE,
    ROUND(AVG(a.MRR), 2) AS ARPU
FROM RAW.CUENTAS a
GROUP BY DATE_TRUNC('MONTH', a.FECHA_REGISTRO)
ORDER BY COHORTE DESC;
```
