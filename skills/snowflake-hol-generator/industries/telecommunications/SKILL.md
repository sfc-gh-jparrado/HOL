### Tabla: USO_SERVICIOS
```sql
CREATE OR REPLACE TABLE RAW.USO (
    ID_USO VARCHAR(20) PRIMARY KEY,
    ID_SUSCRIPTOR VARCHAR(15),
    FECHA_HORA TIMESTAMP,
    TIPO_SERVICIO VARCHAR(20), -- VOZ_SALIENTE, VOZ_ENTRANTE, SMS, DATOS
    DURACION_SEGUNDOS NUMBER,
    VOLUMEN_MB NUMBER(10,2),
    DESTINO VARCHAR(20), -- LOCAL, LARGA_DISTANCIA, INTERNACIONAL, RED_PROPIA, OTRA_RED
    CELDA_ID VARCHAR(20),
    COSTO NUMBER(8,2),
    FOREIGN KEY (ID_SUSCRIPTOR) REFERENCES RAW.SUSCRIPTORES(ID_SUSCRIPTOR)
);

-- Generar uso
INSERT INTO RAW.USO
SELECT 
    'USO' || TO_CHAR(CURRENT_DATE(), 'YYYYMMDD') || LPAD(SEQ4()::VARCHAR, 10, '0') AS ID_USO,
    'SUB' || LPAD(UNIFORM(1, 50000, RANDOM())::VARCHAR, 8, '0') AS ID_SUSCRIPTOR,
    DATEADD('second', -UNIFORM(1, 2592000, RANDOM()), CURRENT_TIMESTAMP()) AS FECHA_HORA,
    ARRAY_CONSTRUCT('VOZ_SALIENTE', 'VOZ_ENTRANTE', 'SMS', 'DATOS', 'DATOS', 'DATOS')[UNIFORM(0, 5, RANDOM())]::VARCHAR AS TIPO_SERVICIO,
    CASE WHEN UNIFORM(0, 5, RANDOM()) < 3 THEN UNIFORM(10, 1800, RANDOM()) ELSE 0 END AS DURACION_SEGUNDOS,
    CASE WHEN UNIFORM(0, 5, RANDOM()) >= 3 THEN ROUND(UNIFORM(1, 500, RANDOM())::FLOAT, 2) ELSE 0 END AS VOLUMEN_MB,
    ARRAY_CONSTRUCT('LOCAL', 'LOCAL', 'LOCAL', 'LARGA_DISTANCIA', 'RED_PROPIA', 'OTRA_RED')[UNIFORM(0, 5, RANDOM())]::VARCHAR AS DESTINO,
    'CEL' || LPAD(UNIFORM(1, 5000, RANDOM())::VARCHAR, 5, '0') AS CELDA_ID,
    ROUND(UNIFORM(0, 50, RANDOM())::FLOAT, 2) AS COSTO
FROM TABLE(GENERATOR(ROWCOUNT => 500000));
```

## Vistas Analíticas

```sql
-- Vista de métricas de suscriptores
CREATE OR REPLACE VIEW ANALYTICS.V_METRICAS_SUSCRIPTORES AS
SELECT 
    p.NOMBRE_PLAN,
    p.TIPO_SERVICIO,
    p.CATEGORIA,
    s.SEGMENTO,
    s.CIUDAD,
    
    -- Base
    COUNT(DISTINCT s.ID_SUSCRIPTOR) AS TOTAL_SUSCRIPTORES,
    COUNT(DISTINCT CASE WHEN s.ESTATUS = 'ACTIVO' THEN s.ID_SUSCRIPTOR END) AS SUSCRIPTORES_ACTIVOS,
    
    -- Churn
    COUNT(DISTINCT CASE WHEN s.ESTATUS IN ('PORTADO', 'BAJA') THEN s.ID_SUSCRIPTOR END) AS BAJAS,
    ROUND(AVG(s.RIESGO_CHURN) * 100, 2) AS RIESGO_CHURN_PROMEDIO,
    
    -- Valor
    SUM(p.PRECIO_MENSUAL) AS INGRESO_POTENCIAL_MENSUAL,
    ROUND(AVG(p.PRECIO_MENSUAL), 2) AS ARPU

FROM RAW.SUSCRIPTORES s
JOIN RAW.PLANES p ON s.ID_PLAN = p.ID_PLAN
GROUP BY p.NOMBRE_PLAN, p.TIPO_SERVICIO, p.CATEGORIA, s.SEGMENTO, s.CIUDAD;

-- Vista de uso de red
CREATE OR REPLACE VIEW ANALYTICS.V_USO_RED AS
SELECT 
    DATE_TRUNC('HOUR', u.FECHA_HORA) AS HORA,
    c.CIUDAD,
    c.TECNOLOGIA,
    
    -- Volumen
    COUNT(DISTINCT u.ID_SUSCRIPTOR) AS USUARIOS_ACTIVOS,
    COUNT(*) AS TOTAL_EVENTOS,
    
    -- Voz
    SUM(CASE WHEN u.TIPO_SERVICIO LIKE 'VOZ%' THEN u.DURACION_SEGUNDOS ELSE 0 END) / 60.0 AS MINUTOS_VOZ,
    
    -- Datos
    SUM(u.VOLUMEN_MB) / 1024.0 AS GB_CONSUMIDOS,
    ROUND(AVG(u.VOLUMEN_MB), 2) AS MB_PROMEDIO_SESION,
    
    -- SMS
    COUNT(CASE WHEN u.TIPO_SERVICIO = 'SMS' THEN 1 END) AS TOTAL_SMS

FROM RAW.USO u
JOIN RAW.CELDAS c ON u.CELDA_ID = c.CELDA_ID
GROUP BY DATE_TRUNC('HOUR', u.FECHA_HORA), c.CIUDAD, c.TECNOLOGIA;

-- Vista de churn risk
CREATE OR REPLACE VIEW ANALYTICS.V_CHURN_RISK AS
SELECT 
    s.ID_SUSCRIPTOR,
    s.NOMBRE,
    s.NUMERO_TELEFONO,
    p.NOMBRE_PLAN,
    s.SEGMENTO,
    s.RIESGO_CHURN,
    DATEDIFF('day', s.FECHA_ULTIMA_RECARGA, CURRENT_DATE()) AS DIAS_SIN_RECARGA,
    
    -- Uso reciente
    (SELECT COUNT(*) FROM RAW.USO u WHERE u.ID_SUSCRIPTOR = s.ID_SUSCRIPTOR 
     AND u.FECHA_HORA >= DATEADD('day', -30, CURRENT_TIMESTAMP())) AS EVENTOS_30_DIAS,
    
    -- Tickets
    (SELECT COUNT(*) FROM RAW.TICKETS t WHERE t.ID_SUSCRIPTOR = s.ID_SUSCRIPTOR 
     AND t.FECHA_CREACION >= DATEADD('day', -90, CURRENT_TIMESTAMP())) AS TICKETS_90_DIAS,
    
    -- Clasificación
    CASE 
        WHEN s.RIESGO_CHURN >= 0.7 THEN '🔴 Crítico'
        WHEN s.RIESGO_CHURN >= 0.5 THEN '🟡 Alto'
        WHEN s.RIESGO_CHURN >= 0.3 THEN '🟠 Medio'
        ELSE '🟢 Bajo'
    END AS NIVEL_RIESGO

FROM RAW.SUSCRIPTORES s
JOIN RAW.PLANES p ON s.ID_PLAN = p.ID_PLAN
WHERE s.ESTATUS = 'ACTIVO'
ORDER BY s.RIESGO_CHURN DESC;

-- Vista de QoS por celda
CREATE OR REPLACE VIEW ANALYTICS.V_QOS_CELDAS AS
SELECT 
    c.CELDA_ID,
    c.NOMBRE_SITIO,
    c.CIUDAD,
    c.TECNOLOGIA,
    c.CAPACIDAD_USUARIOS,
    
    -- Carga
    COUNT(DISTINCT u.ID_SUSCRIPTOR) AS USUARIOS_CONECTADOS,
    ROUND(COUNT(DISTINCT u.ID_SUSCRIPTOR)::FLOAT / NULLIF(c.CAPACIDAD_USUARIOS, 0) * 100, 2) AS UTILIZACION_PCT,
    
    -- Throughput
    ROUND(SUM(u.VOLUMEN_MB) / 1024, 2) AS GB_TOTALES,
    
    -- Estado
    c.ESTATUS,
    CASE 
        WHEN COUNT(DISTINCT u.ID_SUSCRIPTOR)::FLOAT / NULLIF(c.CAPACIDAD_USUARIOS, 0) > 0.9 THEN '🔴 Congestionada'
        WHEN COUNT(DISTINCT u.ID_SUSCRIPTOR)::FLOAT / NULLIF(c.CAPACIDAD_USUARIOS, 0) > 0.7 THEN '🟡 Alta carga'
        ELSE '🟢 Normal'
    END AS ESTADO_CARGA

FROM RAW.CELDAS c
LEFT JOIN RAW.USO u ON c.CELDA_ID = u.CELDA_ID 
    AND u.FECHA_HORA >= DATEADD('day', -7, CURRENT_TIMESTAMP())
WHERE c.ESTATUS = 'ACTIVA'
GROUP BY c.CELDA_ID, c.NOMBRE_SITIO, c.CIUDAD, c.TECNOLOGIA, c.CAPACIDAD_USUARIOS, c.ESTATUS;
```
