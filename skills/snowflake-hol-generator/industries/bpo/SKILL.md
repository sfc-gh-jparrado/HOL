## Vistas Analíticas

```sql
-- Vista de métricas de call center
CREATE OR REPLACE VIEW ANALYTICS.V_METRICAS_CALL_CENTER AS
SELECT 
    DATE_TRUNC('HOUR', i.FECHA_HORA_INICIO) AS HORA,
    ca.NOMBRE_CAMPANA,
    cb.NOMBRE_EMPRESA AS CLIENTE,
    i.CANAL,
    
    -- Volumen
    COUNT(*) AS TOTAL_INTERACCIONES,
    COUNT(CASE WHEN i.TIPO = 'INBOUND' THEN 1 END) AS INBOUND,
    COUNT(CASE WHEN i.TIPO = 'OUTBOUND' THEN 1 END) AS OUTBOUND,
    
    -- Tiempos
    ROUND(AVG(i.DURACION_SEGUNDOS), 0) AS AHT_SEGUNDOS, -- Average Handle Time
    ROUND(AVG(i.TIEMPO_ESPERA_SEGUNDOS), 0) AS ASA_SEGUNDOS, -- Average Speed of Answer
    ROUND(AVG(i.TIEMPO_HOLD_SEGUNDOS), 0) AS HOLD_PROMEDIO,
    
    -- SLA
    ROUND(COUNT(CASE WHEN i.TIEMPO_ESPERA_SEGUNDOS <= cb.SLA_TIEMPO_RESPUESTA THEN 1 END)::FLOAT / 
          NULLIF(COUNT(*), 0) * 100, 2) AS SLA_PCT,
    
    -- FCR
    ROUND(COUNT(CASE WHEN i.RESOLUCION_PRIMER_CONTACTO THEN 1 END)::FLOAT / 
          NULLIF(COUNT(*), 0) * 100, 2) AS FCR_PCT,
    
    -- Transfers
    ROUND(COUNT(CASE WHEN i.TRANSFERIDA THEN 1 END)::FLOAT / 
          NULLIF(COUNT(*), 0) * 100, 2) AS TRANSFER_RATE_PCT

FROM RAW.INTERACCIONES i
JOIN RAW.CAMPANAS ca ON i.ID_CAMPANA = ca.ID_CAMPANA
JOIN RAW.CLIENTES_BPO cb ON ca.ID_CLIENTE_BPO = cb.ID_CLIENTE_BPO
GROUP BY DATE_TRUNC('HOUR', i.FECHA_HORA_INICIO), ca.NOMBRE_CAMPANA, cb.NOMBRE_EMPRESA, i.CANAL;

-- Vista de performance de agentes
CREATE OR REPLACE VIEW ANALYTICS.V_PERFORMANCE_AGENTES AS
SELECT 
    a.ID_AGENTE,
    a.NOMBRE,
    a.NIVEL,
    a.TURNO,
    
    -- Productividad
    COUNT(i.ID_INTERACCION) AS TOTAL_INTERACCIONES,
    ROUND(SUM(i.DURACION_SEGUNDOS) / 3600.0, 2) AS HORAS_PRODUCTIVAS,
    ROUND(AVG(i.DURACION_SEGUNDOS), 0) AS AHT_INDIVIDUAL,
    
    -- Calidad
    ROUND(AVG(CASE WHEN i.RESOLUCION_PRIMER_CONTACTO THEN 100 ELSE 0 END), 2) AS FCR_PCT,
    ROUND(AVG(e.CSAT_SCORE), 2) AS CSAT_PROMEDIO,
    ROUND(AVG(e.NPS_SCORE), 2) AS NPS_PROMEDIO,
    
    -- Clasificación
    CASE 
        WHEN AVG(e.CSAT_SCORE) >= 4.5 AND AVG(CASE WHEN i.RESOLUCION_PRIMER_CONTACTO THEN 100 ELSE 0 END) >= 80 THEN '⭐ Top Performer'
        WHEN AVG(e.CSAT_SCORE) >= 4.0 THEN '✅ Cumple'
        WHEN AVG(e.CSAT_SCORE) >= 3.0 THEN '⚠️ En desarrollo'
        ELSE '🔴 Requiere coaching'
    END AS CLASIFICACION

FROM RAW.AGENTES a
LEFT JOIN RAW.INTERACCIONES i ON a.ID_AGENTE = i.ID_AGENTE
LEFT JOIN RAW.ENCUESTAS e ON i.ID_INTERACCION = e.ID_INTERACCION AND e.COMPLETADA = TRUE
WHERE a.ESTATUS = 'ACTIVO'
GROUP BY a.ID_AGENTE, a.NOMBRE, a.NIVEL, a.TURNO;

-- Vista de NPS por cliente
CREATE OR REPLACE VIEW ANALYTICS.V_NPS_CLIENTES AS
SELECT 
    cb.NOMBRE_EMPRESA,
    COUNT(e.ID_ENCUESTA) AS TOTAL_ENCUESTAS,
    COUNT(CASE WHEN e.NPS_SCORE >= 9 THEN 1 END) AS PROMOTORES,
    COUNT(CASE WHEN e.NPS_SCORE BETWEEN 7 AND 8 THEN 1 END) AS PASIVOS,
    COUNT(CASE WHEN e.NPS_SCORE <= 6 THEN 1 END) AS DETRACTORES,
    ROUND(
        (COUNT(CASE WHEN e.NPS_SCORE >= 9 THEN 1 END)::FLOAT - 
         COUNT(CASE WHEN e.NPS_SCORE <= 6 THEN 1 END)::FLOAT) / 
        NULLIF(COUNT(e.ID_ENCUESTA), 0) * 100, 0
    ) AS NPS_SCORE
FROM RAW.CLIENTES_BPO cb
JOIN RAW.CAMPANAS ca ON cb.ID_CLIENTE_BPO = ca.ID_CLIENTE_BPO
JOIN RAW.INTERACCIONES i ON ca.ID_CAMPANA = i.ID_CAMPANA
JOIN RAW.ENCUESTAS e ON i.ID_INTERACCION = e.ID_INTERACCION
WHERE e.COMPLETADA = TRUE
GROUP BY cb.NOMBRE_EMPRESA;
```

## Análisis con Cortex AI

```sql
-- Análisis de sentimiento de comentarios
CREATE OR REPLACE VIEW ANALYTICS.V_ANALISIS_COMENTARIOS AS
SELECT 
    e.ID_ENCUESTA,
    e.COMENTARIO,
    e.CSAT_SCORE,
    e.NPS_SCORE,
    
    -- Sentimiento del comentario
    ROUND(SNOWFLAKE.CORTEX.SENTIMENT(e.COMENTARIO), 3) AS SENTIMIENTO_SCORE,
    
    -- Clasificación del tema
    SNOWFLAKE.CORTEX.CLASSIFY_TEXT(
        e.COMENTARIO,
        ['Tiempo de espera', 'Calidad de atención', 'Resolución del problema', 'Actitud del agente', 'Proceso complicado', 'Satisfacción general']
    ):label::STRING AS TEMA_PRINCIPAL,
    
    -- Resumen si es largo
    CASE 
        WHEN LENGTH(e.COMENTARIO) > 100 
        THEN SNOWFLAKE.CORTEX.SUMMARIZE(e.COMENTARIO)
        ELSE e.COMENTARIO
    END AS RESUMEN

FROM RAW.ENCUESTAS e
WHERE e.COMENTARIO IS NOT NULL AND e.COMPLETADA = TRUE;
```
