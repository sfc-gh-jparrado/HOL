### Tabla: PRODUCTOS
```sql
CREATE OR REPLACE TABLE RAW.PRODUCTOS (
    ID_PRODUCTO VARCHAR(10) PRIMARY KEY,
    SKU VARCHAR(20),
    NOMBRE_PRODUCTO VARCHAR(200),
    CATEGORIA VARCHAR(50),
    TIEMPO_CICLO_MIN NUMBER(5,2),
    COSTO_PRODUCCION NUMBER(10,2),
    PRECIO_VENTA NUMBER(10,2),
    PESO_KG NUMBER(6,3),
    ACTIVO BOOLEAN DEFAULT TRUE
);

INSERT INTO RAW.PRODUCTOS
SELECT 
    'PRD' || LPAD(SEQ4()::VARCHAR, 3, '0') AS ID_PRODUCTO,
    'SKU-' || UNIFORM(10000, 99999, RANDOM()) AS SKU,
    ARRAY_CONSTRUCT(
        'Componente Electrónico', 'Pieza Plástica', 'Ensamble Completo', 
        'Carcasa', 'Módulo', 'Tarjeta', 'Sensor', 'Conector'
    )[UNIFORM(0, 7, RANDOM())]::VARCHAR || ' ' || UNIFORM(100, 999, RANDOM()) AS NOMBRE_PRODUCTO,
    ARRAY_CONSTRUCT('Electrónicos', 'Plásticos', 'Ensambles', 'Componentes')[UNIFORM(0, 3, RANDOM())]::VARCHAR AS CATEGORIA,
    ROUND(UNIFORM(5, 60, RANDOM())::FLOAT / 10, 2) AS TIEMPO_CICLO_MIN,
    ROUND(UNIFORM(10, 500, RANDOM())::FLOAT, 2) AS COSTO_PRODUCCION,
    ROUND(UNIFORM(20, 800, RANDOM())::FLOAT, 2) AS PRECIO_VENTA,
    ROUND(UNIFORM(1, 100, RANDOM())::FLOAT / 10, 3) AS PESO_KG,
    TRUE AS ACTIVO
FROM TABLE(GENERATOR(ROWCOUNT => 50));
```

### Tabla: ORDENES_PRODUCCION
```sql
CREATE OR REPLACE TABLE RAW.ORDENES_PRODUCCION (
    ID_ORDEN VARCHAR(15) PRIMARY KEY,
    FECHA_CREACION TIMESTAMP,
    FECHA_PROGRAMADA DATE,
    FECHA_INICIO TIMESTAMP,
    FECHA_FIN TIMESTAMP,
    ID_PRODUCTO VARCHAR(10),
    ID_LINEA VARCHAR(10),
    CANTIDAD_PROGRAMADA NUMBER,
    CANTIDAD_PRODUCIDA NUMBER,
    CANTIDAD_DEFECTUOSA NUMBER,
    ESTADO VARCHAR(20), -- PENDIENTE, EN_PROCESO, COMPLETADA, CANCELADA
    PRIORIDAD VARCHAR(10), -- ALTA, MEDIA, BAJA
    FOREIGN KEY (ID_PRODUCTO) REFERENCES RAW.PRODUCTOS(ID_PRODUCTO),
    FOREIGN KEY (ID_LINEA) REFERENCES RAW.LINEAS_PRODUCCION(ID_LINEA)
);

-- Generar órdenes de producción
INSERT INTO RAW.ORDENES_PRODUCCION
SELECT 
    'ORD' || TO_CHAR(CURRENT_DATE(), 'YYYYMMDD') || LPAD(SEQ4()::VARCHAR, 5, '0') AS ID_ORDEN,
    DATEADD('hour', -UNIFORM(1, 2160, RANDOM()), CURRENT_TIMESTAMP()) AS FECHA_CREACION,
    DATEADD('day', -UNIFORM(0, 90, RANDOM()), CURRENT_DATE()) AS FECHA_PROGRAMADA,
    DATEADD('hour', -UNIFORM(1, 2000, RANDOM()), CURRENT_TIMESTAMP()) AS FECHA_INICIO,
    CASE WHEN UNIFORM(1, 10, RANDOM()) <= 8 
         THEN DATEADD('hour', -UNIFORM(1, 1900, RANDOM()), CURRENT_TIMESTAMP()) 
         ELSE NULL END AS FECHA_FIN,
    'PRD' || LPAD(UNIFORM(1, 50, RANDOM())::VARCHAR, 3, '0') AS ID_PRODUCTO,
    'LIN' || LPAD(UNIFORM(1, 8, RANDOM())::VARCHAR, 3, '0') AS ID_LINEA,
    UNIFORM(100, 5000, RANDOM()) AS CANTIDAD_PROGRAMADA,
    UNIFORM(80, 5000, RANDOM()) AS CANTIDAD_PRODUCIDA,
    UNIFORM(0, 50, RANDOM()) AS CANTIDAD_DEFECTUOSA,
    ARRAY_CONSTRUCT('COMPLETADA', 'COMPLETADA', 'COMPLETADA', 'EN_PROCESO', 'PENDIENTE')[UNIFORM(0, 4, RANDOM())]::VARCHAR AS ESTADO,
    ARRAY_CONSTRUCT('ALTA', 'MEDIA', 'MEDIA', 'MEDIA', 'BAJA')[UNIFORM(0, 4, RANDOM())]::VARCHAR AS PRIORIDAD
FROM TABLE(GENERATOR(ROWCOUNT => 2000));
```

## Vistas Analíticas

```sql
-- Vista de eficiencia de producción
CREATE OR REPLACE VIEW ANALYTICS.V_EFICIENCIA_PRODUCCION AS
SELECT 
    o.ID_ORDEN,
    o.FECHA_PROGRAMADA,
    DATE_TRUNC('MONTH', o.FECHA_PROGRAMADA) AS MES,
    
    -- Planta y línea
    pl.NOMBRE_PLANTA,
    l.NOMBRE_LINEA,
    l.TIPO_PRODUCTO,
    
    -- Producto
    p.NOMBRE_PRODUCTO,
    p.CATEGORIA,
    
    -- Métricas de cantidad
    o.CANTIDAD_PROGRAMADA,
    o.CANTIDAD_PRODUCIDA,
    o.CANTIDAD_DEFECTUOSA,
    o.CANTIDAD_PRODUCIDA - o.CANTIDAD_DEFECTUOSA AS CANTIDAD_BUENA,
    
    -- KPIs
    ROUND(o.CANTIDAD_PRODUCIDA::FLOAT / NULLIF(o.CANTIDAD_PROGRAMADA, 0) * 100, 2) AS PCT_CUMPLIMIENTO,
    ROUND((o.CANTIDAD_PRODUCIDA - o.CANTIDAD_DEFECTUOSA)::FLOAT / NULLIF(o.CANTIDAD_PRODUCIDA, 0) * 100, 2) AS PCT_CALIDAD,
    ROUND(o.CANTIDAD_DEFECTUOSA::FLOAT / NULLIF(o.CANTIDAD_PRODUCIDA, 0) * 100, 2) AS PCT_DEFECTOS,
    
    -- Costos
    o.CANTIDAD_PRODUCIDA * p.COSTO_PRODUCCION AS COSTO_TOTAL,
    o.CANTIDAD_DEFECTUOSA * p.COSTO_PRODUCCION AS COSTO_DESPERDICIO,
    
    o.ESTADO,
    o.PRIORIDAD

FROM RAW.ORDENES_PRODUCCION o
JOIN RAW.LINEAS_PRODUCCION l ON o.ID_LINEA = l.ID_LINEA
JOIN RAW.PLANTAS pl ON l.ID_PLANTA = pl.ID_PLANTA
JOIN RAW.PRODUCTOS p ON o.ID_PRODUCTO = p.ID_PRODUCTO;

-- Vista OEE (Overall Equipment Effectiveness)
CREATE OR REPLACE VIEW ANALYTICS.V_OEE_LINEAS AS
SELECT 
    l.ID_LINEA,
    l.NOMBRE_LINEA,
    pl.NOMBRE_PLANTA,
    l.CAPACIDAD_HORA,
    l.VELOCIDAD_ESTANDAR,
    
    -- Disponibilidad (asumiendo 8 horas por turno)
    ROUND(480.0 / 480 * 100, 2) AS DISPONIBILIDAD_PCT,
    
    -- Rendimiento
    ROUND(l.VELOCIDAD_ESTANDAR::FLOAT / NULLIF(l.CAPACIDAD_HORA, 0) * 100, 2) AS RENDIMIENTO_PCT,
    
    -- Calidad (promedio de órdenes)
    (SELECT ROUND(AVG((o.CANTIDAD_PRODUCIDA - o.CANTIDAD_DEFECTUOSA)::FLOAT / NULLIF(o.CANTIDAD_PRODUCIDA, 0) * 100), 2)
     FROM RAW.ORDENES_PRODUCCION o 
     WHERE o.ID_LINEA = l.ID_LINEA AND o.ESTADO = 'COMPLETADA') AS CALIDAD_PCT,
    
    l.ESTADO,
    l.ULTIMO_MANTENIMIENTO,
    DATEDIFF('day', l.ULTIMO_MANTENIMIENTO, CURRENT_DATE()) AS DIAS_DESDE_MANTENIMIENTO

FROM RAW.LINEAS_PRODUCCION l
JOIN RAW.PLANTAS pl ON l.ID_PLANTA = pl.ID_PLANTA;

-- Vista de alertas de inventario
CREATE OR REPLACE VIEW ANALYTICS.V_ALERTAS_MATERIAS AS
SELECT 
    ID_MATERIA,
    NOMBRE_MATERIA,
    PROVEEDOR,
    STOCK_ACTUAL,
    STOCK_MINIMO,
    CASE 
        WHEN STOCK_ACTUAL < STOCK_MINIMO * 0.5 THEN '🔴 CRÍTICO'
        WHEN STOCK_ACTUAL < STOCK_MINIMO THEN '🟡 BAJO'
        ELSE '🟢 OK'
    END AS ESTADO_STOCK,
    LEAD_TIME_DIAS,
    ROUND(STOCK_ACTUAL / NULLIF(STOCK_MINIMO, 0), 2) AS COBERTURA_RATIO
FROM RAW.MATERIAS_PRIMAS
ORDER BY STOCK_ACTUAL / NULLIF(STOCK_MINIMO, 0);
```
