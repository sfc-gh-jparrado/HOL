# Sub-Skill: Sales Analytics

## Metadata
- **Parent**: snowflake-hol-generator
- **Name**: cross-functional/sales-analytics
- **Área**: Ventas / Comercial
- **Aplicable a**: Todas las industrias B2B

## 🎯 Contexto de Negocio

Este módulo transversal agrega capacidades de análisis comercial:
- Pipeline de ventas y oportunidades
- Forecasting y proyecciones
- Performance de vendedores
- Análisis de win/loss

## Vistas Analíticas

```sql
-- Vista de pipeline
CREATE OR REPLACE VIEW ANALYTICS.V_PIPELINE AS
SELECT 
    e.NOMBRE_ETAPA,
    e.ORDEN,
    e.PROBABILIDAD_CIERRE AS PROB_ETAPA,
    COUNT(o.ID_OPORTUNIDAD) AS NUM_OPORTUNIDADES,
    SUM(o.MONTO) AS VALOR_PIPELINE,
    SUM(o.MONTO * e.PROBABILIDAD_CIERRE / 100) AS VALOR_PONDERADO,
    ROUND(AVG(o.MONTO), 2) AS TICKET_PROMEDIO,
    ROUND(AVG(DATEDIFF('day', o.FECHA_CREACION, COALESCE(o.FECHA_CIERRE_REAL, CURRENT_DATE()))), 0) AS DIAS_PROMEDIO_EN_ETAPA
FROM RAW.OPORTUNIDADES o
JOIN RAW.ETAPAS_PIPELINE e ON o.ID_ETAPA = e.ID_ETAPA
WHERE o.ID_ETAPA NOT IN ('ETQ006', 'ETQ007') -- Excluir cerradas
GROUP BY e.NOMBRE_ETAPA, e.ORDEN, e.PROBABILIDAD_CIERRE
ORDER BY e.ORDEN;

-- Vista de forecast
CREATE OR REPLACE VIEW ANALYTICS.V_FORECAST AS
SELECT 
    DATE_TRUNC('MONTH', o.FECHA_CIERRE_ESPERADA) AS MES_CIERRE,
    v.NOMBRE AS VENDEDOR,
    v.REGION,
    v.SEGMENTO,
    v.CUOTA_TRIMESTRAL / 3 AS CUOTA_MENSUAL,
    
    -- Pipeline
    SUM(CASE WHEN o.ID_ETAPA NOT IN ('ETQ006', 'ETQ007') THEN o.MONTO ELSE 0 END) AS PIPELINE_TOTAL,
    SUM(CASE WHEN o.ID_ETAPA NOT IN ('ETQ006', 'ETQ007') THEN o.MONTO * o.PROBABILIDAD / 100 ELSE 0 END) AS FORECAST_PONDERADO,
    
    -- Cerrado
    SUM(CASE WHEN o.ID_ETAPA = 'ETQ006' THEN o.MONTO ELSE 0 END) AS CERRADO_GANADO,
    
    -- % de cuota
    ROUND(SUM(CASE WHEN o.ID_ETAPA = 'ETQ006' THEN o.MONTO ELSE 0 END) / NULLIF(v.CUOTA_TRIMESTRAL / 3, 0) * 100, 2) AS PCT_CUOTA

FROM RAW.OPORTUNIDADES o
JOIN RAW.VENDEDORES v ON o.ID_VENDEDOR = v.ID_VENDEDOR
WHERE o.FECHA_CIERRE_ESPERADA >= DATE_TRUNC('MONTH', CURRENT_DATE())
  AND o.FECHA_CIERRE_ESPERADA < DATEADD('month', 3, DATE_TRUNC('MONTH', CURRENT_DATE()))
GROUP BY DATE_TRUNC('MONTH', o.FECHA_CIERRE_ESPERADA), v.NOMBRE, v.REGION, v.SEGMENTO, v.CUOTA_TRIMESTRAL;

-- Vista de win/loss
CREATE OR REPLACE VIEW ANALYTICS.V_WIN_LOSS AS
SELECT 
    DATE_TRUNC('QUARTER', o.FECHA_CIERRE_REAL) AS TRIMESTRE,
    v.REGION,
    v.SEGMENTO,
    c.INDUSTRIA,
    
    -- Conteos
    COUNT(CASE WHEN o.ID_ETAPA = 'ETQ006' THEN 1 END) AS GANADAS,
    COUNT(CASE WHEN o.ID_ETAPA = 'ETQ007' THEN 1 END) AS PERDIDAS,
    
    -- Montos
    SUM(CASE WHEN o.ID_ETAPA = 'ETQ006' THEN o.MONTO ELSE 0 END) AS VALOR_GANADO,
    SUM(CASE WHEN o.ID_ETAPA = 'ETQ007' THEN o.MONTO ELSE 0 END) AS VALOR_PERDIDO,
    
    -- Win Rate
    ROUND(COUNT(CASE WHEN o.ID_ETAPA = 'ETQ006' THEN 1 END)::FLOAT / 
          NULLIF(COUNT(CASE WHEN o.ID_ETAPA IN ('ETQ006', 'ETQ007') THEN 1 END), 0) * 100, 2) AS WIN_RATE_PCT,
    
    -- Ciclo de ventas
    ROUND(AVG(CASE WHEN o.ID_ETAPA = 'ETQ006' 
              THEN DATEDIFF('day', o.FECHA_CREACION, o.FECHA_CIERRE_REAL) END), 0) AS CICLO_VENTA_DIAS

FROM RAW.OPORTUNIDADES o
JOIN RAW.VENDEDORES v ON o.ID_VENDEDOR = v.ID_VENDEDOR
JOIN RAW.CUENTAS c ON o.ID_CUENTA = c.ID_CUENTA
WHERE o.FECHA_CIERRE_REAL IS NOT NULL
GROUP BY DATE_TRUNC('QUARTER', o.FECHA_CIERRE_REAL), v.REGION, v.SEGMENTO, c.INDUSTRIA;

-- Vista de razones de pérdida
CREATE OR REPLACE VIEW ANALYTICS.V_RAZONES_PERDIDA AS
SELECT 
    o.MOTIVO_PERDIDA,
    o.COMPETIDOR,
    COUNT(*) AS OPORTUNIDADES_PERDIDAS,
    SUM(o.MONTO) AS VALOR_PERDIDO,
    ROUND(AVG(o.MONTO), 2) AS TICKET_PROMEDIO_PERDIDO,
    ROUND(COUNT(*)::FLOAT / NULLIF((SELECT COUNT(*) FROM RAW.OPORTUNIDADES WHERE ID_ETAPA = 'ETQ007'), 0) * 100, 2) AS PCT_DEL_TOTAL
FROM RAW.OPORTUNIDADES o
WHERE o.ID_ETAPA = 'ETQ007'
  AND o.MOTIVO_PERDIDA IS NOT NULL
GROUP BY o.MOTIVO_PERDIDA, o.COMPETIDOR
ORDER BY VALOR_PERDIDO DESC;

-- Vista de performance de vendedores
CREATE OR REPLACE VIEW ANALYTICS.V_PERFORMANCE_VENDEDORES AS
SELECT 
    v.ID_VENDEDOR,
    v.NOMBRE,
    v.REGION,
    v.SEGMENTO,
    v.CUOTA_ANUAL,
    
    -- Resultados YTD
    SUM(CASE WHEN o.ID_ETAPA = 'ETQ006' AND YEAR(o.FECHA_CIERRE_REAL) = YEAR(CURRENT_DATE()) THEN o.MONTO ELSE 0 END) AS CERRADO_YTD,
    ROUND(SUM(CASE WHEN o.ID_ETAPA = 'ETQ006' AND YEAR(o.FECHA_CIERRE_REAL) = YEAR(CURRENT_DATE()) THEN o.MONTO ELSE 0 END) / 
          NULLIF(v.CUOTA_ANUAL, 0) * 100, 2) AS PCT_CUOTA_YTD,
    
    -- Pipeline activo
    SUM(CASE WHEN o.ID_ETAPA NOT IN ('ETQ006', 'ETQ007') THEN o.MONTO ELSE 0 END) AS PIPELINE_ACTIVO,
    
    -- Actividad
    (SELECT COUNT(*) FROM RAW.ACTIVIDADES a WHERE a.ID_VENDEDOR = v.ID_VENDEDOR 
     AND a.FECHA >= DATEADD('day', -30, CURRENT_TIMESTAMP())) AS ACTIVIDADES_30D,
    
    -- Clasificación
    CASE 
        WHEN SUM(CASE WHEN o.ID_ETAPA = 'ETQ006' AND YEAR(o.FECHA_CIERRE_REAL) = YEAR(CURRENT_DATE()) THEN o.MONTO ELSE 0 END) / 
             NULLIF(v.CUOTA_ANUAL, 0) >= 1 THEN '🏆 Top Performer'
        WHEN SUM(CASE WHEN o.ID_ETAPA = 'ETQ006' AND YEAR(o.FECHA_CIERRE_REAL) = YEAR(CURRENT_DATE()) THEN o.MONTO ELSE 0 END) / 
             NULLIF(v.CUOTA_ANUAL, 0) >= 0.8 THEN '✅ En meta'
        WHEN SUM(CASE WHEN o.ID_ETAPA = 'ETQ006' AND YEAR(o.FECHA_CIERRE_REAL) = YEAR(CURRENT_DATE()) THEN o.MONTO ELSE 0 END) / 
             NULLIF(v.CUOTA_ANUAL, 0) >= 0.5 THEN '⚠️ En riesgo'
        ELSE '🔴 Bajo cuota'
    END AS CLASIFICACION

FROM RAW.VENDEDORES v
LEFT JOIN RAW.OPORTUNIDADES o ON v.ID_VENDEDOR = o.ID_VENDEDOR
WHERE v.ESTATUS = 'ACTIVO'
GROUP BY v.ID_VENDEDOR, v.NOMBRE, v.REGION, v.SEGMENTO, v.CUOTA_ANUAL;
```
